```dart
// Requires network access — run manually on a device or emulator with internet.
// Do NOT include in the offline unit test suite (flutter test test/).
// Run with: flutter test integration_test/core_flow_test.dart
//
// Flow: app boots → card loads → save to favourites → Favourites tab shows card in grid.

import 'package:flutter/material.dart';
import 'package:flutter_test/flutter_test.dart';
import 'package:integration_test/integration_test.dart';
import 'package:random_magic/main.dart' as app;
import 'package:skeletonizer/skeletonizer.dart';

void main() {
  IntegrationTestWidgetsFlutterBinding.ensureInitialized();

  testWidgets(
    'core flow: app boots, card loads, save to favourites, card appears in grid (TEST-06)',
    (tester) async {
      // 1. Boot the full app.
      app.main();
      await tester.pumpAndSettle(const Duration(seconds: 5));

      // 2. Assert a card has loaded (Skeletonizer is gone — real card visible).
      // The Skeletonizer enabled flag becomes false once the card resolves.
      expect(
        find.byWidgetPredicate((w) => w is Skeletonizer && w.enabled),
        findsNothing,
        reason: 'Expected card to have loaded (Skeletonizer disabled), but shimmer is still showing.',
      );

      // 3. Assert the bookmark button is visible (success state).
      expect(find.byTooltip('Save to Favourites'), findsOneWidget);

      // 4. Tap the bookmark button to save the card.
      await tester.tap(find.byTooltip('Save to Favourites'));
      await tester.pumpAndSettle(const Duration(seconds: 2));

      // Snackbar 'Saved to Favourites' confirms the save.
      expect(find.text('Saved to Favourites'), findsOneWidget);

      // 5. Navigate to the Favourites tab.
      // The bottom nav uses NavigationDestination. Find by icon or label.
      // GoRouter shell uses NavigationBar — destinations may be labeled 'Favourites'
      // or use Icons.favorite. Adjust finder based on actual app_router.dart labels.
      final favTab = find.byIcon(Icons.favorite);
      if (favTab.evaluate().isNotEmpty) {
        await tester.tap(favTab.first);
      } else {
        // Fallback: find by NavigationDestination label text.
        await tester.tap(find.text('Favourites').last);
      }
      await tester.pumpAndSettle(const Duration(seconds: 2));

      // 6. Assert at least one card appears in the grid.
      // FavouritesScreen uses a SliverGrid of CachedNetworkImage cards.
      // The grid items are tappable — verify at least one exists.
      // GridView or CustomScrollView must have at least 1 child.
      final gridFinder = find.byType(GridView).first;
      expect(gridFinder, findsOneWidget,
          reason: 'Expected FavouritesScreen grid to be visible after saving a card.');
    },
    timeout: const Timeout(Duration(minutes: 2)),
  );
}
```

If the app uses NavigationBar instead of BottomNavigationBar, the tab finder may need adjustment.
Read app_router.dart to determine the exact Favourites tab label/icon before writing the final version.
